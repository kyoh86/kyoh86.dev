【メモ】【OSX】peco/fzf でブランチ切替

git branch -a \
  | perl -ne'm|[^/\* ]+$|;next if$_{$&};$_{$&}=1;next if m,^\*|remotes/.+/HEAD,;$s="\e[0m<S>";s|.*/([^/]+)/|\e[31m$1$s|; s|^ +([^/]+)$|\e[34mlocal$s$1|;print' \
  | column -ts'<S>' \
  | fzf --ansi \
  | xargs git checkout
